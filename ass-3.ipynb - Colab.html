import requests

def get_apod_data(api_key):
    url = f'https://api.nasa.gov/planetary/apod?api_key={api_key}'
    response = requests.get(url)
    data = response.json()
    return data

import dash
import dash_core_components as dcc
import dash_html_components as html

app = dash.Dash(__name__)

def generate_apod_dashboard(data):
    return html.Div([
        html.H1("NASA Astronomy Picture of the Day"),
        html.Img(src=data['url'], style={'width': '50%'}),
        html.H3(data['title']),
        html.P(data['explanation'])
    ])

@app.callback(
    dash.dependencies.Output('apod-dashboard', 'children'),
    [dash.dependencies.Input('refresh-button', 'n_clicks')]
)
def update_apod_dashboard(n_clicks):
    if n_clicks is None:
        n_clicks = 0
    api_key = 'Mpn9gmcA4pZM1dya4eniuVjYDMv0lpYBEEtlS54B'  # Replace with your NASA API key
    data = get_apod_data(api_key)
    return generate_apod_dashboard(data)

app.layout = html.Div([
    html.H2("NASA APOD Dashboard"),
    html.Button("Refresh", id='refresh-button', n_clicks=0),
    html.Div(id='apod-dashboard')
])

if __name__ == '__main__':
    app.run_server(debug=True)
